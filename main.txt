function menu_tree(objId)
{
    document.cookie = "xyjLeftMenu" + "=" + escape(objId) + ";path=/";
}

function getCookie(){
    if(document.cookie)
    {
        var start = document.cookie.indexOf('xyjLeftMenu=');
        if (start == -1) 
        { 
        	return ""; 
        }
        else
        {
        	var end = document.cookie.indexOf(";", start);
        	if (end == -1) 
        	{ 
        		end = document.cookie.length;
        	}
        	end -= start;
        	var cookie = document.cookie.substr(start,end);
        	return unescape(cookie.substr(cookie.indexOf('=') + 1, cookie.length - cookie.indexOf('=') + 1));
        }
    }
}

function initCookie()
{
    var cookie =  getCookie();
    if(cookie != null && cookie != "")
    {
    	expandCookie(cookie);
    }
}

function expandCookie(cookie)
{
	var obj = "#" + cookie;
    $(obj).parent().attr("class","act");
    $(obj).parent().parent().css("display","");
    $(obj).parent().parent().prev().attr("class","active");
}
